O pacote flutter_contacts mudou a API. O código que o Junie botou está errado. Aqui está o código corrigido pra colar:

// lib/services/contacts_service.dart
import 'package:flutter_contacts/flutter_contacts.dart';
class ContactsService {
  Future<bool> requestPermission() async {
    return await FlutterContacts.requestPermission();
  }
  Future<List<ContactResult>> findByName(String query) async {
    final hasPermission = await FlutterContacts.requestPermission();
    if (!hasPermission) return [];
    final allContacts = await FlutterContacts.getContacts();
    final normalizedQuery = _normalize(query);
    final results = <ContactResult>[];
    for (final contact in allContacts) {
      final name = contact.displayName;
      if (name == null || name.isEmpty) continue;
      final normalizedName = _normalize(name);
      final score = _matchScore(normalizedQuery, normalizedName);
      if (score > 0.4) {
        final phone = contact.phones.isNotEmpty
            ? contact.phones.first.number
            : null;
        if (phone == null || phone.isEmpty) continue;
        results.add(ContactResult(
          name: name,
          phone: phone,
          score: score,
        ));
      }
    }
    results.sort((a, b) => b.score.compareTo(a.score));
    return results;
  }
  String _normalize(String input) {
    return input
        .toLowerCase()
        .trim()
        .replaceAll(RegExp(r'[^\w\s]'), '')
        .replaceAll(RegExp(r'\s+'), ' ');
  }
  double _matchScore(String query, String name) {
    if (name == query) return 1.0;
    if (name.contains(query)) return 0.8;
    if (query.split(' ').every((w) => name.contains(w))) return 0.6;
    return 0.0;
  }
}
class ContactResult {
  final String name;
  final String phone;
  final double score;
  ContactResult({required this.name, required this.phone, required this.score});
}
